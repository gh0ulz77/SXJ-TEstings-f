#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>

// OLED
#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
#define OLED_RESET -1
Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);

// Joystick
#define JOY_Y 34
#define JOY_BTN 32

// LEDs
#define LedRed 12
#define LedBlue 14

// Extra GPIO "power"
#define GPIO1 26
#define GPIO2 27

// Touch
#define TOUCH_PIN 4

// Menu
String options[] = {"Red", "Blue", "All", "Off"};
int selected = 0;
const int total = 4;

// Lock
bool locked = true;
bool lastTouch = false;
bool lastBtn = false;

void setup() {
  Serial.begin(115200);
  Wire.begin(21, 22);

  pinMode(JOY_BTN, INPUT_PULLUP);
  pinMode(TOUCH_PIN, INPUT);

  pinMode(GPIO1, OUTPUT);
  pinMode(GPIO2, OUTPUT);
  digitalWrite(GPIO1, HIGH);
  digitalWrite(GPIO2, HIGH);

  pinMode(LedRed, OUTPUT);
  pinMode(LedBlue, OUTPUT);

  if(!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)){
    Serial.println("OLED failed");
    while(1);
  }
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);
  display.display();
}

void loop() {
  int joyY = analogRead(JOY_Y);
  bool btn = digitalRead(JOY_BTN) == LOW;
  bool touch = digitalRead(TOUCH_PIN) == HIGH;

  // 🔐 Toggle lock
  if(touch && !lastTouch){
    locked = !locked;
    delay(150); // debounce
  }
  lastTouch = touch;

  if(!locked){
    // ⬆️ Scroll up
    if(joyY < 1000){
      selected--;
      if(selected < 0) selected = total - 1;
      delay(150);
    }
    // ⬇️ Scroll down
    if(joyY > 3000){
      selected++;
      if(selected >= total) selected = 0;
      delay(150);
    }

    // ✅ Select
    if(btn && !lastBtn){
      runOption(selected);
      delay(200);
    }
  }

  lastBtn = btn;

  drawUI();
  delay(50);
}
void drawUI(){
  display.clearDisplay();

  // Status
  display.setCursor(0,0);
  display.println("Status: " + String(locked ? "LOCKED" : "UNLOCKED"));

  // Menu
  for(int i=0;i<total;i++){
    display.setCursor(10, 16 + i*12);
    display.print(i==selected ? "> " : "  ");
    display.println(options[i]);
  }

  display.display();
}

void runOption(int i){
  Serial.print("Running: "); Serial.println(options[i]);

  switch(i){
    case 0: // Red LED
      digitalWrite(LedRed, HIGH);
      digitalWrite(LedBlue, LOW);
      break;
    case 1: // Blue LED
      digitalWrite(LedRed, LOW);
      digitalWrite(LedBlue, HIGH);
      break;
    case 2: // Both
      digitalWrite(LedRed, HIGH);
      digitalWrite(LedBlue, HIGH);
      break;
    case 3: // Off
      digitalWrite(LedRed, LOW);
      digitalWrite(LedBlue, LOW);
      break;
  }
}
